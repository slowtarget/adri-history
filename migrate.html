Required Firestore security rules

Apply these in the Firebase console → Firestore → Rules:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Test metadata + subcollections: authenticated read/write (for this tool).
    match /tests/{testId} {
      allow read:  if request.auth != null;
      allow write: if request.auth != null;
      match /facts/{factId} {
        allow read:  if request.auth != null;
        allow write: if request.auth != null;
      }
      match /questions/{questionId} {
        allow read:  if request.auth != null;
        allow write: if request.auth != null;
      }
    }

    // Old flat collections: read-only (migrate.html reads these when migrating in-place).
    match /facts/{doc}     { allow read: if request.auth != null; allow write: if false; }
    match /questions/{doc} { allow read: if request.auth != null; allow write: if false; }

    // Results: users can create their own and read their own — nothing else.
    match /results/{doc} {
      allow get, list: if request.auth != null
                       && request.auth.uid == resource.data.userId;
      allow create:    if request.auth != null
                       && request.resource.data.userId == request.auth.uid;
      allow update, delete: if false;
    }

    // Deny everything else.
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
